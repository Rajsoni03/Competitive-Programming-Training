# Next Greater Element - [Practice - Work@Tech](https://workat.tech/problem-solving/practice/next-greater-element)



<br>



Example:1
```
Input: 
5
1 5 2 3 5

Output: 
5 -1 3 5 -1
```
Example:2
```
Input: 
5
3 2 1 4 5

Output: 
4 4 4 5 -1
```
Example:3
```
Input: 
4
4 3 2 1

Output: 
-1 -1 -1 -1
```

## Solution 1  

Time - O(N<sup>2<sup>)<br>
Space - O(1)

```cpp
vector<int> getNextGreaterElement (vector<int> &A) {
	// add your logic here
	int n = A.size();
	vector<int> ans(n);
	bool notFount;
	for (int i = 0; i < n-1; i++){
		notFount = true;
		for (int j = i+1; j < n; j++){
			if (A[i] < A[j]){
				ans[i] = A[j];
				notFount = false;
				break;
			}
		}
		if (notFount){
			ans[i] = -1;
		}
	}
	ans[n-1] = -1;
	return ans;
}
```

## Solution 2 

Time - O(N)<br>
Space - O(N)

```cpp
vector<int> getNextGreaterElement (vector<int> &A) {
	// add your logic here
	stack<int> s;
	int n = A.size();
	vector<int> ans(n);

	for (int i = n-1; i >= 0; i--){
		while(!s.empty() && s.top() <= A[i]){
			s.pop();			
		}
		if (s.empty()){
			ans[i] = -1;
		}
		else{
			ans[i] = s.top();
		}
		s.push(A[i]);
	}
	return ans;
}
```
